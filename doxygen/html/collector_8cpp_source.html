401 //-----------------------------------------------------------------------------
402 // Receive an event menu item or bouton tool for actualize state machine
403 // then execute 'List' or 'Extract' or ...
404 //
405 // Called by :
406 // 1. Collector::OnMenuListPrj(wxCommandEvent& _event):1,
407 // 2. Collector::OnMenuExtractPrj(wxCommandEvent& _event):1,
408 // 3. Collector::OnMenuListExtractPrj(wxCommandEvent& _event):1,
409 //
410 // 4. Collector::OnMenuListWS(wxCommandEvent& _event):1,
411 // 5. Collector::OnMenuExtractWS(wxCommandEvent& _event):1,
412 // 6. Collector::OnMenuListExtractWS(wxCommandEvent& _event):1,
413 //
414 // Call to :
415 // 1. Collector::callActions(wxCommandEvent& _event):1,
416 //
417 void Collector::OnMenuToState(wxCommandEvent& _event)
418 {
419 printD("=> Begin 'Collector::OnMenuToState(...)'");
420 
421  colState State = fbNone;
422 // origin
423  wxUint64 id = _event.GetId();
424 // only good id's
425  bool goodId = true;
426 
427  if (id == idMbarStop || id == idTbarStop)
428  State = fbStop;
429  else
430  if (id == idMbarCleanTemp || id == idTbarCleanTemp)
431  State = fbCleanTemp;
432  else
433  if (id == idMbarListPrj || id == idTbarListPrj)
434  State = fbListPrj;
435  else
436  // is a macro state : 'fbListPrj' followed by 'fbExtractPrj'
437  if (id == idMbarListExtractPrj || id == idTbarListExtractPrj)
438  State = fbListExtractPrj;
439  else
440  if (id == idMbarExtractPrj || id == idTbarExtractPrj)
441  State = fbExtractPrj;
442  else
443  if (id == idMbarListWS || id == idTbarListWS)
444  State = fbListWS;
445  else
446  if (id == idMbarExtractWS || id == idTbarExtractWS)
447  State = fbExtractWS;
448  else
449  if (id == idMbarListExtractWS || id == idTbarListExtractWS)
450  // is a macro state : 'fbListWS' followed by 'fbExtractWS'
451  State = fbListExtractWS;
452  else
453  goodId = false;
454 
455  if (goodId)
456  {
457  // update state graph
458  actualizeGraphState(State);
459  // actions according state machine
460  callActions(_event);
461  }
462 
463 printD(" <= End 'Collector::OnMenuToState(...)'");
464
