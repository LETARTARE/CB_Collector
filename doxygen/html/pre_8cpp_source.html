253  wxMenu* menu = nullptr;
254  size_t pos = 0;
255  while (true)
256  {
257  // ignore consecutive slashes of '_menuPath'
258  while (pos < _menuPath.Length() && _menuPath.GetChar(pos) == '/')
259  ++pos;
260  // find next slash
261  size_t nextPos = pos;
262  while (nextPos < _menuPath.Length() && _menuPath.GetChar(++nextPos) != '/') ;
263  // 'nextPos'
264  wxString current = _menuPath.Mid(pos, nextPos - pos);
265 //printWarn("... " + quote(current));

267  // the last

269  // current holds the current search string
270  if (!menu) // no menu yet? look in menubar
271  {
272  wxInt32 menuPos = mbar->FindMenu(current);
273  if (menuPos == wxNOT_FOUND) break; // failed
274  else menu = mbar->GetMenu(menuPos);
275  }
276  else
277  {
278  if (isLast)
279  {
280  wxInt32 id = menu->FindItem(current);
281  if (id != wxNOT_FOUND)
282  {

284  wxCommandEvent evt(wxEVT_COMMAND_MENU_SELECTED, id);
285  mbar->GetEventHandler()->ProcessEvent(evt);

289  }

293  if (existing != wxNOT_FOUND) menu = menu->GetMenuItems()[existing]->GetSubMenu();
294  else break; // failed
295  }
